DROP DATABASE IF EXISTS DBPELICULAS;
CREATE DATABASE DBPELICULAS;
USE DBPELICULAS;

CREATE TABLE TB_CATEGORIAS(
COD_CATEGORIA INT (4) NOT NULL,
DESCRIPCION VARCHAR (50) NOT NULL,
PRIMARY KEY (COD_CATEGORIA)
);

CREATE TABLE TB_PELICULAS(
COD_PELICULA INT (4) NOT NULL,
NOM_PELICULA VARCHAR (50) NOT NULL,
DURACION INT (3) NOT NULL,
DIRECTOR VARCHAR (100) NOT NULL,
ORIGEN VARCHAR (50) NOT NULL,
CALIFICACION_EDAD VARCHAR (3) NOT NULL,
COD_CATEGORIA INT (4) NOT NULL,
PRIMARY KEY (COD_PELICULA),
FOREIGN KEY (COD_CATEGORIA) REFERENCES TB_CATEGORIAS(COD_CATEGORIA)
);
